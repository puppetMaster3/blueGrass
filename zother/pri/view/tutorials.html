        <html><head></head><body>
        <form>
            First Name<input type='text' id='firstname' ><br>
            Last Name<input type='text' id='lastname' ><br>
            <button id='lookup' onclick='return false'>Look Up</button>
            <button id='create' onclick='return false'>Create Name</button>
        </form>

        <!-- app -->
        <script  src='http://scdn.primus.netdna-cdn.com/latest/cloudAPI2.js'></script>
    

        <script>
        <!-- insert -->
        cAPI=new CloudAPI()
            var cBut = document.getElementById('create')
            cBut.addEventListener('click', function(e) {
                var firstname = $('#firstname').val()
                var lastname = $('#lastname').val()
                console.log('clicked ' + firstname +', ' + lastname)
                var ename =new Object()
                ename.firstname=firstname
                ename.lastname=lastname
                cAPI.insert('nameapp',ename, onInserted)
            })

            function onInserted(data){
                console.log('back')
                //console.log(data)
            }
    

            <!-- select -->
            var sBut = document.getElementById('lookup')
            sBut.addEventListener('click', function(e) {
                var firstname = $('#firstname').val()
                console.log('clicked ' + firstname)
                var ename =new Object()
                ename.firstname=firstname
                cAPI.select('nameapp', ename, onSelect)
            })

            function onSelect(data){
                console.log(data.array_)
                $('#lastname').val(data.array_[0].lastname)
            }

        </script>
        </body></html>
